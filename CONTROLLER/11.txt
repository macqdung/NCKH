genres = ["manga", "light novel", "kinh dị", "trinh thám", "giả tưởng", "tiểu thuyết", "lãng mạn", "khoa học viễn tưởng", "phiêu lưu", "hồi hộp", "IT", "Finance", "Life skills", "Psychology", "Romance", "Science Fiction"]
moods = ["vui", "buồn", "thư giãn", "hào hứng", "căng thẳng", "tò mò", "lãng đãng", "nghiêm túc", "happy", "sad", "neutral"]
levels = ["mới bắt đầu", "trung cấp", "nâng cao", "kinh nghiệm", "beginner", "intermediate", "advanced"]

# Basic intents and templates
base_templates = {
    "greet": ["chào", "xin chào", "hello", "hi", "good morning", "good evening", "hey"],
    "goodbye": ["tạm biệt", "bye", "hẹn gặp lại", "see you", "chào tạm biệt"],

    "book_query": ["Bạn có thể gợi ý vài cuốn sách?", "Hãy đề xuất 3 cuốn sách cho tôi", "Tôi cần đề xuất sách", "Gợi ý sách theo sở thích của tôi"],

    "smalltalk_food": ["Bạn thích ăn gì?", "Gợi ý món ăn", "Ăn gì ngon?"],

# Expand NLU examples to ~5000
nlu_examples = {intent: list(set(templates)) for intent, templates in base_templates.items()}

# Function to create variations
def vary_phrase(p):
    variants = [p]
    if random.random() < 0.4:
        variants.append(p + " nhé")
    if random.random() < 0.2:
        variants.append("Bạn ơi, " + p)
    if random.random() < 0.15:
        variants.append(p + "?")
    if random.random() < 0.1:
        variants.append(p.upper())
    # small replacements
    if "bạn" in p and random.random() < 0.1:
        variants.append(p.replace("bạn", "các bạn"))
    return random.choice(list(dict.fromkeys(variants)))

# Seed more examples
target_nlu = 5000
all_intents = list(nlu_examples.keys())
while sum(len(v) for v in nlu_examples.values()) < target_nlu:
    intent = random.choice(all_intents)
    base = random.choice(base_templates[intent])
    phrase = vary_phrase(base)
    # sometimes inject entity mentions
    if intent == "provide_genre" and random.random() < 0.4:
        g = random.choice(genres)
        phrase = random.choice([f"Tôi thích {g}", f"{g} nhé", f"Thể loại {g}"])
    if intent == "provide_mood" and random.random() < 0.3:
        m = random.choice(moods)
        phrase = random.choice([f"Tôi đang {m}", m])
    if intent == "provide_level" and random.random() < 0.3:
        l = random.choice(levels)
        phrase = random.choice([f"Tôi là {l}", l])
    nlu_examples[intent].append(phrase)

# Build nlu.yml structure with limited per-intent duplicates
nlu_yaml = {"version": "3.1", "nlu": []}
for intent, phrases in nlu_examples.items():
    unique = list(dict.fromkeys(phrases))
    # add simple entity annotations for provide_genre/mood/level
    if intent in ("provide_genre",):
        examples_block = []
        for p in unique[:2000]:
            # annotate genre if present
            found = None
            for g in genres:
                if g in p.lower():
                    found = g
                    break
            if found:
                examples_block.append(f"- {p.replace(found, f'[{found}](genre)')}")
            else:
                examples_block.append(f"- {p}")
        nlu_yaml["nlu"].append({"intent": intent, "examples": "\n".join(examples_block)})
    elif intent in ("provide_mood",):
        examples_block = []
        for p in unique[:2000]:
            found = None
            for m in moods:
                if m in p.lower():
                    found = m
                    break
            if found:
                examples_block.append(f"- {p.replace(found, f'[{found}](mood)')}")
            else:
                examples_block.append(f"- {p}")
        nlu_yaml["nlu"].append({"intent": intent, "examples": "\n".join(examples_block)})
    elif intent in ("provide_level",):
        examples_block = []
        for p in unique[:2000]:
            found = None
            for l in levels:
                if l in p.lower():
                    found = l
                    break
            if found:
                examples_block.append(f"- {p.replace(found, f'[{found}](level)')}")
            else:
                examples_block.append(f"- {p}")
        nlu_yaml["nlu"].append({"intent": intent, "examples": "\n".join(examples_block)})
    else:
        nlu_yaml["nlu"].append({"intent": intent, "examples": "\n".join(f"- {p}" for p in unique[:1000])})

# Save nlu.yml
nlu_path = OUT / "nlu.yml"
with nlu_path.open("w", encoding="utf-8") as f:
    yaml.dump(nlu_yaml, f, allow_unicode=True)

# Generate stories ~3000
stories = {"version": "3.1", "stories": []}
story_count = 3000
for i in range(1, story_count+1):
    steps = []
    # optionally greet
    if random.random() < 0.2:
        steps.append({"intent": "greet"})
        if random.random() < 0.6:
            steps.append({"action": "utter_greet"})
    # random flow types
    flow = random.choices(
        ["probe_genre_then_query", "probe_mood_then_query", "complete_profile_then_query", "direct_query", "smalltalk_then_query"],
        weights=[25,25,20,20,10], k=1
    )[0]
    if flow == "probe_genre_then_query":
        steps.append({"intent": "ask_genre"})
        steps.append({"action": "utter_ask_genre"})
        # user provides
        steps.append({"intent": "provide_genre"})
        # bot recommends
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "probe_mood_then_query":
        steps.append({"intent": "ask_mood"})
        steps.append({"action": "utter_ask_mood"})
        steps.append({"intent": "provide_mood"})
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "complete_profile_then_query":
        steps.append({"intent": "provide_genre"})
        steps.append({"intent": "provide_mood"})
        steps.append({"intent": "provide_level"})
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "direct_query":
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    else:
        steps.append({"intent": random.choice(["smalltalk_weather","smalltalk_feeling","smalltalk_daily_life"])})
        steps.append({"action": random.choice(["utter_smalltalk_weather","utter_smalltalk_feeling","utter_smalltalk_daily_life"])})
        # then user asks for books
        if random.random() < 0.7:
            steps.append({"intent": "book_query"})
            steps.append({"action": "utter_recommend_books"})
    # sometimes thanks/goodbye
    if random.random() < 0.2:
        steps.append({"intent": "thanks"})
        steps.append({"action": "utter_thanks"})
    if random.random() < 0.1:
        steps.append({"intent": "goodbye"})
        steps.append({"action": "utter_goodbye"})
    # convert to story block
    story_block = {"story": f"story_{i}", "steps": []}
    for s in steps:
        if "intent" in s:
            story_block["steps"].append({"intent": s["intent"]})
        elif "action" in s:
            story_block["steps"].append({"action": s["action"]})
    stories["stories"].append(story_block)

stories_path = OUT / "stories.yml"
with stories_path.open("w", encoding="utf-8") as f:
    yaml.dump(stories, f, allow_unicode=True)

            steps.append({"action": "utter_greet"})
    flow = random.choices(
        ["probe_genre_then_query", "probe_mood_then_query", "complete_profile_then_query", "direct_query", "smalltalk_then_query"],
        weights=[25,25,20,20,10], k=1
    )[0]
    if flow == "probe_genre_then_query":
        steps.append({"intent": "ask_genre"})
        steps.append({"action": "utter_ask_genre"})
        steps.append({"intent": "provide_genre"})
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "probe_mood_then_query":
        steps.append({"intent": "ask_mood"})
        steps.append({"action": "utter_ask_mood"})
        steps.append({"intent": "provide_mood"})
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "complete_profile_then_query":
        steps.append({"intent": "provide_genre"})
        steps.append({"intent": "provide_mood"})
        steps.append({"intent": "provide_level"})
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    elif flow == "direct_query":
        steps.append({"intent": "book_query"})
        steps.append({"action": "utter_recommend_books"})
    else:
        steps.append({"intent": random.choice(["smalltalk_weather","smalltalk_feeling","smalltalk_daily_life"])})
        steps.append({"action": random.choice(["utter_smalltalk_weather","utter_smalltalk_feeling","utter_smalltalk_daily_life"])})
        if random.random() < 0.7:
            steps.append({"intent": "book_query"})
            steps.append({"action": "utter_recommend_books"})
    if random.random() < 0.2:
        steps.append({"intent": "thanks"})
        steps.append({"action": "utter_thanks"})
    if random.random() < 0.1:
        steps.append({"intent": "goodbye"})
        steps.append({"action": "utter_goodbye"})
    story_block = {"story": f"story_{i}", "steps": []}
    for s in steps:
        if "intent" in s:
            story_block["steps"].append({"intent": s["intent"]})
        elif "action" in s:
            story_block["steps"].append({"action": s["action"]})
    stories["stories"].append(story_block)

stories_path = OUT / "stories.yml"
with stories_path.open("w", encoding="utf-8") as f:
    yaml.dump(stories, f, allow_unicode=True)
